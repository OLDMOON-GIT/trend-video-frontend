 */
export async function GET(request: NextRequest) {
  try {
    const user = await getCurrentUser(request);
    if (!user) {
      return NextResponse.json(
        { error: 'Î°úÍ∑∏?∏Ïù¥ ?ÑÏöî?©Îãà??' },
        { status: 401 }
      );
    }

    // ?¨Ïö©?êÏùò ?ºÎ∏îÎ¶¨Ïãú???ÅÌíà Ï°∞Ìöå
    const products = db.prepare(`
      SELECT
        id, title, description, category,
        original_price, discount_price,
        image_url, deep_link
      FROM coupang_products
      WHERE user_id = ? AND status = 'published'
      ORDER BY created_at DESC
    `).all(user.userId) as any[];

    if (products.length === 0) {
      return NextResponse.json(
        { error: '?ºÎ∏îÎ¶¨Ïãú???ÅÌíà???ÜÏäµ?àÎã§.' },
        { status: 404 }
      );
    }

    // HTML ?ùÏÑ±
    const html = generateStaticHTML(products);

    return NextResponse.json({
      success: true,
      html,
      productCount: products.length
    });

  } catch (error: any) {
    console.error('??HTML ?ùÏÑ± ?§Î•ò:', error);
    return NextResponse.json(
      { error: error?.message || 'HTML ?ùÏÑ± ?§Ìå®' },
      { status: 500 }
    );
  }
}

/**
 * HTML ?¥Ïä§ÏºÄ?¥ÌîÑ (XSS Î∞©Ï?)
 */
function escapeHtml(text: string): string {
  const map: { [key: string]: string } = {
    '&': '&amp;',
    '<': '&lt;',
    '>': '&gt;',
    '"': '&quot;',
    "'": '&#039;'
  };
  return text.replace(/[&<>"']/g, m => map[m]);
}
